1. SELECT `nazwa`, `dlugosc`, `szerokosc` FROM `smok` WHERE `pochodzenie` = "Polska";
2. SELECT DISTINCT `pochodzenie` FROM `smok` ORDER BY `pochodzenie` DESC; 
3. SELECT parada.rok, ROUND(AVG(smok.dlugosc)) as "Średnia długość" FROM `parada` INNER JOIN smok ON smok.id = parada.id WHERE parada.rok > "2005" GROUP BY parada.rok;
