1. SELECT LOWER(`nazwa`) FROM `wojewodztwa`;
2. SELECT COUNT(`id`) as "Ilość miast" FROM `miasta` WHERE `id_wojewodztwa` = 1;
3. SELECT miasta.nazwa, wojewodztwa.nazwa FROM `miasta`INNER JOIN wojewodztwa ON wojewodztwa.id = miasta.id_wojewodztwa WHERE miasta.nazwa LIKE "Lu%" ORDER BY `miasta`.`nazwa`;
4. SELECT wojewodztwa.nazwa, COUNT(miasta.nazwa) AS "Liczba miast" FROM wojewodztwa JOIN miasta ON wojewodztwa.id = miasta.id_wojewodztwa GROUP BY wojewodztwa.nazwa;
