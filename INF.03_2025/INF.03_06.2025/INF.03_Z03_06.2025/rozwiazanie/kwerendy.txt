1. SELECT `nazwisko`, `imie`, `klasa`, `data_wypozyczenia` FROM `uczniowie` WHERE `data_wypozyczenia` > "2020-03-02";
2. SELECT `imie`, `nazwisko`, `data_wypozyczenia`, `autor`, `kod` FROM `uczniowie` INNER JOIN `ksiazki` ON ksiazki.id = uczniowie.id_ksiazki;
3. DELETE FROM `ksiazki` WHERE `autor` = "0";
4. SELECT `autor`, `tytul`, `kod` FROM `ksiazki` ORDER BY RAND() LIMIT 5;
